DER (texto) - ConectaUNI

TABELA: users
  - id BIGINT PK
  - full_name VARCHAR(255)
  - email VARCHAR(255) UNIQUE
  - password_hash VARCHAR(128)
  - salt VARCHAR(64)
  - role VARCHAR(32)
  - created_at TIMESTAMP

TABELA: events
  - id BIGINT PK
  - title VARCHAR(255)
  - description TEXT
  - start_at DATETIME
  - end_at DATETIME
  - location VARCHAR(512)
  - budget DOUBLE
  - creator_id BIGINT FK -> users(id)

TABELA: invites
  - id BIGINT PK
  - event_id BIGINT FK -> events(id)
  - invitee_email VARCHAR(255)
  - token VARCHAR(255) UNIQUE
  - qr_code_path VARCHAR(1024)
  - rsvp_status VARCHAR(32)
  - created_at TIMESTAMP

TABELA: evaluations
  - id BIGINT PK
  - event_id BIGINT FK -> events(id)
  - user_id BIGINT FK -> users(id)
  - stars INT
  - comment TEXT
  - created_at TIMESTAMP
